10. Vad kommer utskriften att bli? (@property och @<metod>.setter) ​

python
class Cirkeln:
    def __init__(self, radie):
        self._radie = radie

    @property
    def radie(self):
        return self._radie

    @radie.setter
    def radie(self, värde):
        if värde > 0:
            self._radie = värde

cirkel = Cirkeln(10)
cirkel.radie = 15
print(cirkel.radie)
