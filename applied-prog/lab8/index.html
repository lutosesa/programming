Laboration: DC Motor ​

Inledning ​

I detta labb kommer vi koppla till Arduino Uno en likström-motor och en NPN-transistor. En NPN-transistor behövs för att styra en DC-motor med en Arduino eftersom en Arduino inte kan leverera tillräckligt med ström för att driva motorn direkt.
Här är en enkel förklaring:

  1. Arduino ger inte tillräckligt med ström: En DC-motor kräver vanligtvis mer ström än vad en Arduino kan leverera direkt via sina digitala utgångar (pins). Arduino kan bara ge runt 40 mA per pin, medan en DC-motor ofta kräver mycket mer ström (150 mA).

  2. Transistorn fungerar som en strömbrytare: En NPN-transistor används som en strömbrytare som kan hantera högre ström för att driva motorn. När Arduinon skickar en liten ström till transistorens bas (styrpinne), låter transistorn en mycket större ström passera genom kollektorn och emittern till motorn, vilket gör att motorn kan snurra.

  3. Säkerhet för Arduino: Genom att använda en transistor skyddar du också Arduino från att skadas av den höga ström som motorn kräver. Istället för att belasta Arduino direkt, använder du transistorn som en mellanhand för att koppla Arduino till motorn.

Syfte med labbet ​

Du kommer att lära dig om:

  • hur man kopplar in en dc-motor, tryckknappar, transistor och diod.
  • hur man kan styra dessa elektroniska komponenter genom att koda fram en algoritm på utvecklingsmiljön Arduino IDE.
  • hur används de huvudfunktionerna, funktioner som ingår i programmet och skapa egna funktioner och anropa dem.

Vad är DC motor (Direct Current Motor) i arduino? ​

En DC-motor är en likströmsmotor som du kan styra med hjälp av en Arduino.

  • DC-motor betyder att den drivs av likström (ström som går i en riktning).
  • Du kan använda Arduino för att starta, stoppa och ändra hastighet eller riktning på motorn genom att skriva kod som styr strömmen till motorn.

Motorns rpm står för Revolutions Per Minute, vilket betyder varv per minut på svenska. Det anger hur många gånger motorns axel snurrar runt sin egen axel under en minut.

  • Högre rpm betyder att motorn snurrar snabbare.
  • Lägsta rpm innebär att motorn snurrar långsammare.

Vad är en NPN-transistor i Arduino? ​

En NPN-transistor i Arduino är en liten elektronisk komponent som fungerar som en strömbrytare. Den hjälper Arduino att styra större strömmar till saker som motorer eller lampor, eftersom Arduino själv inte kan hantera så mycket ström.

Den har tre ben: bas, kollektor, och emitter:

  • Bas: Får en liten ström från Arduino.
  • Kollector: Där den stora strömmen kommer in (till exempel från en strömkälla).
  • Emitter: Där strömmen går ut och går vidare till exempelvis en motor.

Så här fungerar det:

  1. När en liten ström skickas från Arduino till basen, tillåter transistorn att en större ström flödar från kollektorn till emittern.
  2. Om det inte finns någon ström till basen, är transistorn "avstängd", och ingen ström kan flöda genom kollektorn och emittern.

Elektroniska delar som behövs ​

  • Arduino Uno
  • USB A-to-B Kabel
  • Kopplingsplatta
  • (1) DC-motor
  • (1) Transistor
  • (1) 1 kΩ
  • (2) tryckknappar
  • sladdar

/**************************************************************
 * Labb 8 - Övning 1 - DC-Motor
 * 
 * MOTORSPINNING
 * Detta labb kräver att du kör motorn med en NPN-transistor. 
 * Arduino kan leverera cirka 40 mA ström per stift och 
 * en motor kräver upp till 150 mA. 
 * Transistorn förstärker strömmen
 **************************************************************/

// Anslut transistorns bas till stift 9.
const int MOTOR_PIN = 9;

void setup()
{
    pinMode(MOTOR_PIN, OUTPUT);  // ställer in stiftet som en OUTPUT
    Serial.begin(9600);          // initiera seriell kommunikation
}


void loop()
{
    int onTime  = 3000;  // millisekunder för att slå på motorn
    int offTime = 3000;  // millisekunder för att stänga av motorn

    analogWrite(MOTOR_PIN, 255); // slå på motorn (full hastighet)  
    delay(onTime);               // väntetid för onTime millisekunder 
    analogWrite(MOTOR_PIN, 0);   // stäng av motorn
    delay(offTime);              // väntetid för offTime milliseconds

    // Testa funktionerna nedan EN I TAGET.
    // Ta bort kommentarerna för att testa funktionerna nedan

    speedUpAndDown(); 
    //serialSpeed();
}


/***********************************************************
 * Denna funktion accelererar upp motorn från 0 till full 
 * hastighet, för att sedan accelererar ned till 0.
 ***********************************************************/
void speedUpAndDown()
{
    int speed;
    int delayTime = 1; // millisekunder mellan varje hastighetssteg

    // accelererar motorn
    for(speed = 0; speed <= 255; speed++)
    {
        analogWrite(MOTOR_PIN,speed); // ställer in denSerial.print("speed: ");
        delay(delayTime);             // väntetid mellan hastighetssteg 

        Serial.print("speed: ");
        Serial.println(speed);
    }
    
    for(speed = 255; speed >= 0; speed--)
    {
        analogWrite(MOTOR_PIN,speed); // ställer in den nya hastigheten 
        delay(delayTime);             // väntetid mellan hastighetssteg

        Serial.print("speed: ");
        Serial.println(speed);
    }
}

/********************************************************
 * Funktionen uppmanar användaren att ange en hastighet 
 * på 0-255 genom den seriella monitorn så att motorn 
 * sedan roterar genom det angivna värdet. 
 ********************************************************/
void serialSpeed()
{
    int speed;

    // Detta meddelande nedan skrivs ut bara en gång på monitorn.
    Serial.println("Skriv en hastighet (0-255) i rutan ovan,");
    Serial.println("klicka sedan pa [skicka/send] eller tryck pa [retur]");
    Serial.println();  //Skriv ut en tom rad 

    // "true" är alltid sant, så detta kommer att slinga för alltid. 
    while(true)  
    {
        // Kontrollerar om inkommande data är tillgänglig: 
        while (Serial.available() > 0)
        {
            // parseInt() läser in det första heltalsvärdet från Serial Monitor.
            speed = Serial.parseInt();
            
            // begränsar hastigheten mellan 0 och 255 eftersom analogWrite() 
            // bara fungerar i detta intervall. 
            speed = constrain(speed, 0, 255); 

            //skriver ut den hastighet som du angav på monitorn. 
            Serial.print("Du staller in hastighet till:  ");  
            Serial.println(speed);

            analogWrite(MOTOR_PIN, speed);  // ställer in motorns hastighet. 
        }
    }
}

Denna övning kommer vi att simulera ett löpandeband som transporterar paket från ena sidan av fabriken till den andra. Med en krets som består av en knapp för att starta remmen och en slutstegssensor för att stoppa löpandebandet och kunna lossa paketen.
Målen är:

  • Starta löpandebandet endast med startknappen.
  • Stoppa körningen av löpandebandet när föraren anser vara relevant med knappen, eller
  • Stoppa löpandebandet när förpackningen når slutet av löpandebandet

/******************************************************************
 * Labb 8 - Övning 2 - DC-Motor
 * 
 * LÖPANDE BAND
 * Detta labb simulerar ett löpandeband som transporterar paket 
 * från ena sidan av fabriken till den andra. Med en krets som 
 * består av en knapp för att starta remmen och en slutstegssensor 
 * för att stoppa löpandetbandet och kunna lossa paketen.
 * Målen är:
 *  - Starta löpandebandet endast med startknappen.
 *  - Stoppa körningen av löpandebandet när föraren anser vara 
 *    relevant med knappen, eller
 *	- toppa löpandebandet när förpackningen når slutet av 
 *    löpandebandet. 
 ******************************************************************/
 
const int PIN_MOTOR = 9; 
const int PIN_START_MOTOR = 6; 
const int PIN_STOP_MOTOR  = 3;

bool state = 0;

void setup()
{
  	pinMode(PIN_STOP_MOTOR,  INPUT_PULLUP); 
  	pinMode(PIN_START_MOTOR, INPUT_PULLUP); 
	pinMode(PIN_MOTOR, OUTPUT); 
}

void loop()
{
    int startValue = digitalRead(PIN_START_MOTOR); //start-knappen tillstånd läses av. 
    int stopValue  = digitalRead(PIN_STOP_MOTOR);  //stopp-knappen tillstånd läses av. 

    if (startValue == LOW)  //Om tillståndet på "startValue" är lågt, 0. 
    { 					   
      	delay(500); 
      	state = 1;  //ändras tillståndet till 1 för att aktivera motorn. 
    }

    if (stopValue == LOW)   //Om tillståndet på "stopValue" är lågt, 0.
    { 		                        
      	state = 0; 	//ändras tillståndet till 0 för att stänga av motorn 
    } 

    if(state == 0)  // Om tillståndet för "state" är 0, stängs av motorn. 
    {
      	digitalWrite (PIN_MOTOR, LOW);
    }
    else   // annars sätts motorn på.			  
    { 
      	digitalWrite (PIN_MOTOR, HIGH); 
    }
}

Övning 3 ​

Nu ska du ansluta två lysdioder och en piezobuzzer. Du ska göra följande:

  • När motorn är i vila ska den gröna lysdioden vara tänd.
  • När start-knappen blir intryck, körs igång motorn och den röda lysdioden ska tändas och den gröna släckas, samt buzzern ska avge ett kort ljud i början av motorns start för att sedan släckas medans motorn fortsätter att köra.
  • När stop-knappen blir intryck ska motorn stanna, och den gröna lysdioden tändas och den röda släckas.
  • När start-knappen blir intryck igen under exekveringen, körs igång motorn igen och den röda lysdioden tänds igen och den gröna släcks, samt buzzern avger ett kort ljud igen i början av motorns start för att sedan släckas medans motorn fortsätter att köra.

3.2 Programkod ​

cpp
/************************************************************
 * Labb 8 - Övning 3 - DC-Motor
 * 
 * LÖPANDE BAND
 * Detta labb simulerar ett löpandeband som transporterar 
 * paket från ena sidan av fabriken till den andra. Med 
 * en krets som består av en knapp för att starta remmen 
 * och en slutstegssensor för att stoppa löpandetbandet 
 * och kunna lossa paketen.
 * Målen är:
 *  - Starta löpandebandet endast med startknappen, röd 
      lampa tänds och ett ljud avges.
 *  - Stoppa löpandebandet när det behövs med stopp-knappen.
      En grön lampa tänds. 
 *************************************************************/

const int PIN_MOTOR       = 9; 
const int PIN_STOP_MOTOR  = 3;
const int PIN_START_MOTOR = 4;
const int PIN_LED_GREEN   = 5;
const int PIN_LED_RED     = 6;
const int PIN_BUZZER      = A1;

bool state = 0;
bool buzzerPlaying = true;

void setup()
{
    pinMode(PIN_STOP_MOTOR,  INPUT_PULLUP); 
    pinMode(PIN_START_MOTOR, INPUT_PULLUP); 
    pinMode(PIN_MOTOR,      OUTPUT);
    pinMode(PIN_LED_GREEN,  OUTPUT);
    pinMode(PIN_LED_RED,    OUTPUT);
}

void loop()
{
    int startValue = digitalRead(PIN_START_MOTOR); //start-knappen tillstånd läses av.  
    int stopValue  = digitalRead(PIN_STOP_MOTOR);  //stopp-knappen tillstånd läses av. 
   
    if (startValue == LOW)  //Om tillståndet på "startValue" är lågt, 0. 
    {             
        delay(500); 
        state = 1;     //ändras tillståndet till 1 för att aktivera motorn. 
    }
  
    if (stopValue == LOW) //Om tillståndet på "stopValue" är lågt, 0.
    {                             
        state = 0;  //ändras tillståndet till 0 för att stänga av motorn 
    }
    
  
    if(state == 0)  // Om tillståndet för "state" är 0, stängs av motorn. 
    {
        digitalWrite (PIN_LED_GREEN, HIGH);
        digitalWrite (PIN_LED_RED, LOW);
        digitalWrite (PIN_MOTOR, LOW); 
        buzzerPlaying = true; 
    }
    else  // annars sätts motorn på.        
    {
        digitalWrite (PIN_MOTOR, HIGH);
        digitalWrite (PIN_LED_GREEN, LOW);
        digitalWrite (PIN_LED_RED, HIGH);
        
        if( buzzerPlaying == true )
        {
            playBuzzer();
            buzzerPlaying = false; 
        }    
    }
}


void playBuzzer()
{
    tone(PIN_BUZZER, 490); //Skapar en ton(ljud)
    delay(450);
    noTone(PIN_BUZZER);    //stänger av ljudet.
    delay(150);
    tone(PIN_BUZZER, 490); //Skapar en ton(ljud)
    delay(350);
    noTone(PIN_BUZZER);    //stänger av ljudet.
    delay(100);
    tone(PIN_BUZZER, 490); //Skapar en ton(ljud)
    delay(250);
    noTone(PIN_BUZZER);    //stänger av ljudet.
}
